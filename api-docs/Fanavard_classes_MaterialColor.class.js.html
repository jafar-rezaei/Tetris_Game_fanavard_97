    /**
     * Pick a random property of object
     * @param obj
     * @return {*}
     */
    static pickRandomProperty(obj) {
        let result,count = 0;
        for (let prop in obj)
            if (Math.random() < 1 / ++count) result = prop;
        return result;
    }

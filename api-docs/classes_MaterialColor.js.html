	/**
     * Pick a random property of object
     * @param obj
     * @return HexColorString {string}
     */
	static pickRandomProperty(obj) {
		let result, count = 0;
		for (const prop in obj) if (Math.random() < 1 / ++count) result = prop;
		return result;
	}

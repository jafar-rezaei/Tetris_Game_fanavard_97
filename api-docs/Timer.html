let timer = new Timer({
              onStart: function(){
                  TetrisGame.initValues.paused = false;
              },
              workerOnMessage:function (event) {
                  Storage.set('seconds', event.data);
              },
              onPause:function () {
                  TetrisGame.initValues.paused = true;
              },
              onResume:function () {
                  TetrisGame.initValues.paused = false;

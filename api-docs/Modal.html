Constructor of modal class

Example
//Create a new instance
 let modal = new Modal({
      header : "Our Title goes here" ,
      content : "It is a message from our game to say yes you loose game !",
      buttons : [
          {
              text : "Ok",
              isOk : true,
              onclick : function () {
                  this.destroy();
              }
          },
          {
              text : "Do not",
              notOk : true,
              onclick : function () {
                  alert("say why?!");
              }
          }
      ]
  }, true );

 //Show modal
 modal.show();

 //Destory instance
 modal.destroy();
